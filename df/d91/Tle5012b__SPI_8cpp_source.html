174  void Tle5012b_SPI::sendReceiveSpi(uint16_t* sent_data, uint16_t size_of_sent_data, uint16_t* received_data, uint16_t size_of_received_data)
175  {
176  uint32_t data_index = 0;
177  //send via TX
178  sendConfig();
179  digitalWrite(mCS, LOW);
180 
181  for(data_index = 0; data_index < size_of_sent_data; data_index++)
182  {
183  XMC_SPI_CH_Transmit(m3Wire.channel, sent_data[data_index], XMC_SPI_CH_MODE_STANDARD);
184  while((XMC_SPI_CH_GetStatusFlag(m3Wire.channel) & XMC_SPI_CH_STATUS_FLAG_TRANSMIT_SHIFT_INDICATION) == 0U);
185  XMC_SPI_CH_ClearStatusFlag(m3Wire.channel, XMC_SPI_CH_STATUS_FLAG_TRANSMIT_SHIFT_INDICATION);
186  while (XMC_USIC_CH_GetReceiveBufferStatus(m3Wire.channel) == 0U);
187  received_data[0] = XMC_SPI_CH_GetReceivedData(m3Wire.channel);
188  }
189 
190  // receive via RX
191  receiveConfig();
192  delayMicroseconds(5);
193 
194  for(data_index = 0; data_index < size_of_received_data; data_index++)
195  {
196  XMC_SPI_CH_Transmit(m3Wire.channel, 0x0000, XMC_SPI_CH_MODE_STANDARD);
197  while((XMC_SPI_CH_GetStatusFlag(m3Wire.channel) & XMC_SPI_CH_STATUS_FLAG_TRANSMIT_SHIFT_INDICATION) == 0U);
198  XMC_SPI_CH_ClearStatusFlag(m3Wire.channel, XMC_SPI_CH_STATUS_FLAG_TRANSMIT_SHIFT_INDICATION);
199  while (XMC_USIC_CH_GetReceiveBufferStatus(m3Wire.channel) == 0U);
200  received_data[data_index] = XMC_SPI_CH_GetReceivedData(m3Wire.channel);
201  }
202 
203  digitalWrite(mCS, HIGH);
204  }
