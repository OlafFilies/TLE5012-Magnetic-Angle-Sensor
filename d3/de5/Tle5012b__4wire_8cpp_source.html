165 errorTypes Tle5012b_4wire::checkSafety(uint16_t safety, uint16_t command, uint16_t* readreg, uint16_t length)
166 {
167  errorTypes errorCheck ;
168 
169  if (!((safety) & SYSTEM_ERROR_MASK))
170  {
171  errorCheck = SYSTEM_ERROR;
172  }
173 
174  else if (!((safety) & INTERFACE_ERROR_MASK))
175  {
176  errorCheck = INTERFACE_ACCESS_ERROR;
177  }
178 
179  else if (!((safety) & INV_ANGLE_ERROR_MASK))
180  {
181  errorCheck = INVALID_ANGLE_ERROR;
182  }
183 
184  else
185  {
186  uint16_t lengthOfTemp = length*2 + 2;
187  uint8_t temp[lengthOfTemp];
188 
189  temp[0] = _getFirstByte(command);
190  temp[1] = _getSecondByte(command);
191 
192  for (uint16_t i = 0; i<length; i++)
193  {
194  temp[2 + 2*i] = _getFirstByte(readreg[i]);
195  temp[2 + 2*i+1] = _getSecondByte(readreg[i]);
196  }
197 
198  uint8_t crcReceivedFinal = _getSecondByte(safety);
199 
200  uint8_t crc = _crcCalc(temp, lengthOfTemp);
201 
202  if (crc == crcReceivedFinal)
203  {
204  errorCheck = NO_ERROR;
205  }
206  else
207  {
208  errorCheck = CRC_ERROR;
209  resetSafety();
210  }
211  }
212 
213  return errorCheck;

627 errorTypes Tle5012b_4wire::regularCrcUpdate()
628 {
629  readBlockCRC();
630 
631  uint8_t temp[16];
632 
633  for (uint8_t i = 0; i < CRC_NUM_REGISTERS; i++)
634  {
635  temp[2*i] = _getFirstByte(_registers[i]);
636  temp[(2*i)+1] = _getSecondByte(_registers[i]);
637  }
638 
639  uint8_t crc = _crcCalc(temp, 15);
640 
641  uint16_t firstTempByte = (uint16_t) temp[14];
642  uint16_t secondTempByte = (uint16_t) crc;
643  uint16_t valToSend = (firstTempByte << 8) | secondTempByte;
644 
645  _registers[7] = valToSend;
646 
647  return writeTempCoeffUpdate(valToSend);
648 }
